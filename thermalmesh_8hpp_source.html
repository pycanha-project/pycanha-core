165 [[nodiscard]] bool is_valid() const {
166 // Check that dir1_mesh and dir2_mesh have at least 2 elements, that
167 // they start at 0 and end at 1.0 and that they are sorted.
168 // We also check that 2 * dir1_mesh_size * dir2_mesh_size is less than
169 // MeshIndex (uint32_t) max value to avoid overflow.
170 const bool mesh_valid =
171 (2 * _dir1_mesh.size() * _dir2_mesh.size() <=
172 std::numeric_limits<MeshIndex>::max()) &&
173 (_dir1_mesh.size() >= 2) && (_dir2_mesh.size() >= 2) &&
174 (std::abs(_dir1_mesh[0]) <= LENGTH_TOL) &&
175 (std::abs(_dir1_mesh[_dir1_mesh.size() - 1] - 1.0) <= LENGTH_TOL) &&
176 (std::abs(_dir2_mesh[0]) <= 0.0) &&
177 (std::abs(_dir2_mesh[_dir2_mesh.size() - 1] - 1.0) <= LENGTH_TOL) &&
178 (std::is_sorted(_dir1_mesh.begin(), _dir1_mesh.end())) &&
179 (std::is_sorted(_dir2_mesh.begin(), _dir2_mesh.end()));
180
181 return mesh_valid;
182 }
