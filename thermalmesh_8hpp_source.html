167  [[nodiscard]] bool is_valid() const {
168  // Check that dir1_mesh and dir2_mesh have at least 2 elements, that
169  // they start at 0 and end at 1.0 and that they are sorted.
170  // We also check that 2 * dir1_mesh_size * dir2_mesh_size is less than
171  // MeshIndex (uint32_t) max value to avoid overflow.
172  const bool mesh_valid =
173  (2 * _dir1_mesh.size() * _dir2_mesh.size() <=
174  std::numeric_limits<MeshIndex>::max()) &&
175  (_dir1_mesh.size() >= 2) && (_dir2_mesh.size() >= 2) &&
176  (std::abs(_dir1_mesh[0]) <= LENGTH_TOL) &&
177  (std::abs(_dir1_mesh[_dir1_mesh.size() - 1] - 1.0) <= LENGTH_TOL) &&
178  (std::abs(_dir2_mesh[0]) <= 0.0) &&
179  (std::abs(_dir2_mesh[_dir2_mesh.size() - 1] - 1.0) <= LENGTH_TOL) &&
180  (std::is_sorted(_dir1_mesh.begin(), _dir1_mesh.end())) &&
181  (std::is_sorted(_dir2_mesh.begin(), _dir2_mesh.end()));
182 
183  return mesh_valid;
184  }
