302  void sort_triangles() {
303  // Create a vector of indices
304  // TODO: Change indices to Eigen object (for avoid castings)
305  std::vector<MeshIndex> indices(
306  static_cast<MeshIndex>(_triangles.rows()));
307 
308  if (_triangles.rows() == 0) {
309  return;
310  }
311 
312  std::iota(indices.begin(), indices.end(), 0);
313 
314  // Sort the indices according to the face ids
315  std::sort(indices.begin(), indices.end(),
316  [this](MeshIndex i1, MeshIndex i2) {
317  return _face_ids[i1] < _face_ids[i2];
318  });
319 
320  // Sort the triangles
321  TrianglesList sorted_triangles(_triangles.rows(), _triangles.cols());
322  for (VectorIndex i = 0; i < indices.size(); ++i) {
323  sorted_triangles.row(static_cast<Index>(i)) =
324  _triangles.row(indices[i]);
325  }
326  _triangles = std::move(sorted_triangles);
327 
328  // Reorder the face ids
329  FaceIdsList sorted_face_ids(_face_ids.size());
330  for (Index i = 0; i < static_cast<Index>(indices.size()); ++i) {
331  sorted_face_ids[i] =
332  _face_ids[indices[static_cast<VectorIndex>(i)]];
333  // std::cout << sorted_face_ids[i] << "\n";
334  }
335 
336  _face_ids = std::move(sorted_face_ids);
337  _are_triangles_sorted = true;
338  }

340  void compute_areas() {
341  // Cummulative areas only make sense if the triangles are sorted
342  if (!_are_triangles_sorted) {
343  sort_triangles();
344  }
345 
346  // Size of triangles list
347  auto N = static_cast<MeshIndex>(_triangles.rows());
348 
349  if (N == 0) {
350  return;
351  }
352 
353  // Resize _cumareas to hold N elements
354  _cumareas.resize(N);
355 
356  MeshIndex current_face = _face_ids[0]; // Should be 0
357  double cum_area = 0.0;
358  for (MeshIndex i = 0; i < N; ++i) {
359  // If we are on a new face, reset the cumulative area
360  if (_face_ids[i] != current_face) {
361  current_face = _face_ids[i];
362  cum_area = 0.0;
363  }
364 
365  // Get vertices of triangle i
366  const Point3D p1 = _vertices.row(_triangles(i, 0));
367  const Point3D p2 = _vertices.row(_triangles(i, 1));
368  const Point3D p3 = _vertices.row(_triangles(i, 2));
369 
370  // Compute area of triangle i
371  cum_area += 0.5 * ((p2 - p1).cross(p3 - p1)).norm();
372 
373  // Store cumulative area
374  _cumareas[i] = static_cast<float>(cum_area);
375  }
376  }

468  // The Face ID identifies a face of the thermal mesh. Because there is a

475  // an example of face_ids could be:

513  static_cast<MeshIndex>(trimesh.get_vertices().rows());

521  auto current_n_triangles = static_cast<MeshIndex>(_triangles.rows());
522  auto current_n_edges = static_cast<MeshIndex>(_edges.size());
523

526

528  VerticesListFloat temp_vertices(current_n_points + new_trimesh_n_points,
529  3);
530  temp_vertices << _vertices, trimesh.get_vertices().cast<float>();
531  _vertices = std::move(temp_vertices);
532 
533  // 2) Aggregate triangles with offset
534  TrianglesList temp_triangles(
535  current_n_triangles + new_trimesh_n_triangles, 3);
536  temp_triangles << _triangles,
537  (trimesh.get_triangles().array() + current_n_points);
538  _triangles = std::move(temp_triangles);
539

543  // necessary.
544  auto last_face_id_it =
545  std::max_element(_face_ids.begin(), _face_ids.end());
546  MeshIndex last_face_id = 0;
547  if (last_face_id_it != _face_ids.end()) {
548  last_face_id = *last_face_id_it + 2;
549  }
550  // This check is not necessary by design.
551  // TODO(PERFORMANCE): when sure it works, and use an assert instead.

555  // << "\n";
556  throw std::runtime_error(
557  "Faces IDs numbering error. Contact developers.");
558  }

560  FaceIdsList temp_face_ids(current_n_triangles +
561  new_trimesh_n_triangles);
562  // Copy the current face ids
563  std::copy(_face_ids.begin(), _face_ids.end(), temp_face_ids.begin());

565

568  temp_face_ids.begin() + current_n_triangles,
569  [last_face_id](MeshIndex val) { return val + last_face_id; });
570  _face_ids = std::move(temp_face_ids);

577  temp_areas.begin());
578  std::copy(trimesh.get_cumareas().begin(), trimesh.get_cumareas().end(),
579  temp_areas.begin() + current_n_triangles);

581  // 5) Face activity
582  // TODO(FEATURE)

585  // TODO(FEATURE)

588  _n_faces += trimesh.get_number_of_faces();

591  // 8) Colors
592  _front_colors.push_back(trimesh.get_surface1_color());
593  _back_colors.push_back(trimesh.get_surface2_color());
594 
595  // 9) Geometries start and end indices
596  _geometries_vertices.push_back(_geometries_vertices.back() +

598  _geometries_triangles.push_back(_geometries_triangles.back() +

608  // 10) Add geometry id
609  _geometries_id.push_back(geometry_id);

612  EdgesList adjusted_edges = trimesh.get_edges();
613  for (auto& edge : adjusted_edges) {
614  // Use std::transform to increment each element in the collection
615  std::transform(edge.begin(), edge.end(), edge.begin(),
616  [current_n_points](auto index) {
617  return index + current_n_points;
618  });
619  }
620  _edges.insert(_edges.end(), adjusted_edges.begin(),
621  adjusted_edges.end());
622 
623  // 12) Aggregate perimeter_edges
624  Edges adjusted_perimeter_edges = trimesh.get_perimeter_edges();
625 
626  // for index: adjusted_perimeter_edges) { index += current_n_edges; }
627  std::transform(
628  adjusted_perimeter_edges.begin(), adjusted_perimeter_edges.end(),
629  adjusted_perimeter_edges.begin(),
630  [current_n_edges](auto index) { return index + current_n_edges; });
631 
632  // Calculate the total size needed
633  const Index previous_number_of_perimeter_edges =
634  _perimeter_edges.rows();
635

637  _perimeter_edges.conservativeResize(previous_number_of_perimeter_edges +
638  adjusted_perimeter_edges.rows(),
639  Eigen::NoChange);
640 
641  // Use Eigen's block operation to copy the elements from
642  // adjusted_perimeter_edges into the correct position in
643  // _perimeter_edges
644  _perimeter_edges.block(previous_number_of_perimeter_edges, 0,
645  adjusted_perimeter_edges.rows(), 1) =
646  adjusted_perimeter_edges;
647 
648  // 12) Aggregate faces_edges
649  FaceEdges adjusted_faces_edges = trimesh.get_faces_edges();
650  for (auto& face : adjusted_faces_edges) {
651  // for index: face) { index += current_n_edges; }
652  std::transform(face.begin(), face.end(), face.begin(),
653  [current_n_edges](auto index) {
654  return index + current_n_edges;
655  });
656  }

660  << "\n";
661  _faces_edges.insert(_faces_edges.end(), adjusted_faces_edges.begin(),
662  adjusted_faces_edges.end());
663  }

665  TriMesh get_geometry_mesh(MeshIndex mesh_idx) {
666  // Create TriMesh object corresponding to the mesh of the geometry with
667  // id mesh_idx Check that the mesh_idx is valid
668  if (mesh_idx >= _n_geometries) {
669  throw std::invalid_argument("Invalid mesh index");
670  }
671 
672  // Get the start and end indices of the vertices, triangles and edges
673  const MeshIndex start_vertices = _geometries_vertices[mesh_idx];
674  const MeshIndex end_vertices = _geometries_vertices[mesh_idx + 1];
675  const MeshIndex start_triangles = _geometries_triangles[mesh_idx];
676  const MeshIndex end_triangles = _geometries_triangles[mesh_idx + 1];
677  const MeshIndex start_edges = _geometries_edges[mesh_idx];
678  const MeshIndex end_edges = _geometries_edges[mesh_idx + 1];
679  const MeshIndex start_perimeter_edges =
680  _geometries_perimeter_edges[mesh_idx];
681  const MeshIndex end_perimeter_edges =
682  _geometries_perimeter_edges[mesh_idx + 1];
683 
684  // Copy the content to new objects
685  // Vertices
686  VerticesListDouble new_vertices =
687  _vertices.block(start_vertices, 0, end_vertices - start_vertices, 3)
688  .cast<double>();
689  // Triangles
690  TrianglesList new_triangles = _triangles.block(
691  start_triangles, 0, end_triangles - start_triangles, 3);
692  new_triangles.array() -= start_vertices;
693  // Faces id
694  FaceIdsList new_face_ids = _face_ids.block(
695  start_triangles, 0, end_triangles - start_triangles, 1);
696 
697  // Edges
698  EdgesList new_edges(_edges.begin() + start_edges,
699  _edges.begin() + end_edges);
700  for (auto& edge : new_edges) {
701  // for index: edge) { index -= start_vertices; }
702  std::transform(edge.begin(), edge.end(), edge.begin(),
703  [start_vertices](auto index) {
704  return index - start_vertices;
705  });
706  }
707 
708  // Perimeter edges
709  auto new_number_of_perimeter_edges =
710  end_perimeter_edges - start_perimeter_edges;
711  Edges new_perimeter_edges(new_number_of_perimeter_edges);
712  // Use Eigen's block operation to select the subrange and then apply a
713  // unary operation
714  new_perimeter_edges =
715  _perimeter_edges
716  .segment(start_perimeter_edges, new_number_of_perimeter_edges)
717  .unaryExpr(
718  [start_edges](MeshIndex n) { return n - start_edges; });
719 
720  // Faces edges
721  // TODO(PERFORMANCE): Not needed if faces are sorted
722  const MeshIndex face_idx_start =
723  *(std::min_element(new_face_ids.begin(), new_face_ids.end())) / 2;
724  const MeshIndex face_idx_end =
725  *(std::max_element(new_face_ids.begin(), new_face_ids.end())) / 2;
726 
727  FaceEdges new_faces_edges(_faces_edges.begin() + face_idx_start,
728  _faces_edges.begin() + face_idx_end + 1);
729  std::cout << "New faces edges: " << new_faces_edges.size() << '\n';
730  std::cout << "faces_idx_start: " << face_idx_start << '\n';
731  std::cout << "faces_idx_end: " << face_idx_end + 1 << '\n';
732 
733  // Correct the idx of the edges
734  for (auto& face_edges : new_faces_edges) {
735  // for edge: face_edges) { edge -= start_edges; }
736  std::transform(
737  face_edges.begin(), face_edges.end(), face_edges.begin(),
738  [start_edges](auto edge) { return edge - start_edges; });
739  }
740 
741  // Color
742  const ColorRGB surf_1_color = _front_colors[mesh_idx];
743  const ColorRGB surf_2_color = _back_colors[mesh_idx];
744 
745  // Create the TriMesh object
746  TriMesh geometry_mesh;
747  geometry_mesh.set_vertices(std::move(new_vertices));
748  geometry_mesh.set_triangles(std::move(new_triangles));
749  geometry_mesh.set_face_ids(std::move(new_face_ids));
750  geometry_mesh.set_edges(std::move(new_edges));
751  geometry_mesh.set_perimeter_edges(std::move(new_perimeter_edges));
752  geometry_mesh.set_faces_edges(std::move(new_faces_edges));
753  geometry_mesh.set_surface1_color(surf_1_color);
754  geometry_mesh.set_surface2_color(surf_2_color);
755  return geometry_mesh;

859 inline TriMesh create_2d_rectangular_mesh(const Eigen::VectorXd& dir1_mesh,
860  const Eigen::VectorXd& dir2_mesh,
861  double max_distance_points_dir1,
862  double max_distance_points_dir2) {
863  // Assert preconditions (only in debug mode)
864  assert(dir1_mesh.size() >= 2);
865  assert(dir2_mesh.size() >= 2);
866  assert(2 * dir1_mesh.size() * dir2_mesh.size() <=
867  std::numeric_limits<MeshIndex>::max());
868  assert(utils::is_sorted(dir1_mesh));
869  assert(utils::is_sorted(dir2_mesh));
870 
871  // TODO(PERFORMANCE): Use Eigen instead of std::vector to avoid casting
872 
873  // 1. Determine the number of points to reserve space
874  auto dir1_size = static_cast<MeshIndex>(dir1_mesh.size());
875  auto dir2_size = static_cast<MeshIndex>(dir2_mesh.size());
876 
877  MeshIndex num_points_dir1 = dir1_size; // Minimum number of points
878  MeshIndex num_points_dir2 = dir2_size; // Minimum number of points
879 
880  std::vector<MeshIndex> additional_points_dir1(dir1_size - 1, 0);
881  std::vector<MeshIndex> additional_points_dir2(dir2_size - 1, 0);
882 
883  // Now check if additional points are needed based on
884  // max_distance_points If max_distance is negative or zero, then we
885  // don't add additional points
886  if (max_distance_points_dir1 > LENGTH_TOL) {
887  for (MeshIndex i = 0; i < dir1_size - 1; ++i) {
888  const double distance = dir1_mesh[i + 1] - dir1_mesh[i];
889  const auto num_additional_points = static_cast<MeshIndex>(
890  std::floor(distance / max_distance_points_dir1));
891  num_points_dir1 += num_additional_points;
892  additional_points_dir1[i] = num_additional_points;
893  }
894  }
895 
896  if (max_distance_points_dir2 > LENGTH_TOL) {
897  for (MeshIndex i = 0; i < dir2_size - 1; ++i) {
898  const double distance = dir2_mesh[i + 1] - dir2_mesh[i];
899  const auto num_additional_points = static_cast<MeshIndex>(
900  std::floor(distance / max_distance_points_dir2));
901  num_points_dir2 += num_additional_points;
902  additional_points_dir2[i] = num_additional_points;
903  }
904  }
905 
906  // 2. Create the mesh points (3D points with z = 0)
907  VerticesList points(num_points_dir1 * num_points_dir2, 3);
908  std::vector<double> full_dir1_mesh(
909  static_cast<VectorIndex>(num_points_dir1));
910  std::vector<double> full_dir2_mesh(
911  static_cast<VectorIndex>(num_points_dir2));
912 
913  // Fill the full mesh arrays with the original and additional mesh
914  // points
915  MeshIndex p_idx = 0;
916  for (MeshIndex i_dir1 = 0; i_dir1 < dir1_size - 1; ++i_dir1) {
917  const double distance = (dir1_mesh[i_dir1 + 1] - dir1_mesh[i_dir1]) /
918  (additional_points_dir1[i_dir1] + 1);
919  for (MeshIndex i_add1 = 0; i_add1 <= additional_points_dir1[i_dir1];
920  ++i_add1) {
921  full_dir1_mesh[p_idx] =
922  dir1_mesh[i_dir1] + static_cast<double>(i_add1) * distance;
923  ++p_idx;
924  }
925  }
926  full_dir1_mesh[full_dir1_mesh.size() - 1] = dir1_mesh[dir1_size - 1];
927  p_idx = 0;
928  for (MeshIndex i_dir2 = 0; i_dir2 < dir2_size - 1; ++i_dir2) {
929  const double distance = (dir2_mesh[i_dir2 + 1] - dir2_mesh[i_dir2]) /
930  (additional_points_dir2[i_dir2] + 1);
931  for (MeshIndex i_add1 = 0; i_add1 <= additional_points_dir2[i_dir2];
932  ++i_add1) {
933  full_dir2_mesh[p_idx] =
934  dir2_mesh[i_dir2] + static_cast<double>(i_add1) * distance;
935  ++p_idx;
936  }
937  }
938  full_dir2_mesh[full_dir2_mesh.size() - 1] = dir2_mesh[dir2_size - 1];
939 
940  // Fill the points array
941  p_idx = 0;
942  for (MeshIndex i_dir2 = 0; i_dir2 < num_points_dir2; ++i_dir2) {
943  for (MeshIndex i_dir1 = 0; i_dir1 < num_points_dir1; ++i_dir1) {
944  points(p_idx, 0) = full_dir1_mesh[static_cast<VectorIndex>(i_dir1)];
945  points(p_idx, 1) = full_dir2_mesh[static_cast<VectorIndex>(i_dir2)];
946  points(p_idx, 2) = 0.0;
947  ++p_idx;
948  }
949  }

952 
953  // Reserve space for the edges
954  EdgesList edges((dir1_size - 1) * (dir2_size) +
955  (dir1_size) * (dir2_size - 1));
956 
957  // Fill the edges
958 
959  // First the edges in direction 1
960  MeshIndex e_idx = 0;
961  MeshIndex skip_dir2_rows = 0;
962  for (MeshIndex i_dir2 = 0; i_dir2 < dir2_size; ++i_dir2) {

964  for (MeshIndex i_dir1 = 0; i_dir1 < dir1_size - 1; ++i_dir1) {
965  const MeshIndex num_edge_points =
966  additional_points_dir1[i_dir1] + 2;

968 #if defined(__GNUC__)
969 #pragma GCC diagnostic push
970 #pragma GCC diagnostic ignored "-Wnull-dereference"
971 #endif
972  // num_edge_points where a compiled constant, it would be fine
973  edge[0] = p_idx;
974 #if defined(__GNUC__)
975 #pragma GCC diagnostic pop
976 #endif
977 
978  for (MeshIndex i_add1 = 0; i_add1 <= additional_points_dir1[i_dir1];
979  ++i_add1) {
980  edge[i_add1 + 1] = p_idx + i_add1 + 1;
981  }
982  edges[e_idx] = edge;
983  e_idx++;

985  }
986  if (i_dir2 < dir2_size - 1) {
987  skip_dir2_rows += additional_points_dir2[i_dir2];
988  }
989  }
990

992  MeshIndex skip_dir1_cols = 0;
993  for (MeshIndex i_dir1 = 0; i_dir1 < dir1_size; ++i_dir1) {
994  p_idx = i_dir1 + skip_dir1_cols;
995  for (MeshIndex i_dir2 = 0; i_dir2 < dir2_size - 1; ++i_dir2) {
996  const MeshIndex num_edge_points =
997  additional_points_dir2[i_dir2] + 2;
998  Edges edge(num_edge_points);
999 #if defined(__GNUC__)
1000 #pragma GCC diagnostic push
1001 #pragma GCC diagnostic ignored "-Wnull-dereference"
1002 #endif
1003  // num_edge_points where a compiled constant, it would be fine
1004  edge[0] = p_idx;

1006 #pragma GCC diagnostic pop

1009  ++i_add2) {
1010  edge[i_add2 + 1] = p_idx + (i_add2 + 1) * num_points_dir1;
1011  }
1012  edges[e_idx] = edge;
1013  e_idx++;
1014  p_idx += (num_edge_points - 1) * num_points_dir1;
1015  }
1016  if (i_dir1 < dir1_size - 1) {
1017  skip_dir1_cols += additional_points_dir1[i_dir1];
1018  }
1019  }
1020 
1021  // 4. Create the permiter edges. Anti-clockwise order starting from P01
1022  // -> dir1 edge
1023  const MeshIndex num_perimeter_edges = 2 * (dir1_size + dir2_size - 2);
1024  EdgesIdsList perimeter_edges(num_perimeter_edges);
1025  MeshIndex per_edge_idx = 0;
1026  for (MeshIndex i_dir1 = 0; i_dir1 < dir1_size - 1; ++i_dir1) {
1027  perimeter_edges[per_edge_idx] = i_dir1;
1028  ++per_edge_idx;
1029  }
1030  MeshIndex skip_horizontal_edges = (dir2_size) * (dir1_size - 1);
1031  for (MeshIndex i_dir2 = 0; i_dir2 < dir2_size - 1; ++i_dir2) {
1032  perimeter_edges[per_edge_idx] =
1033  i_dir2 + (dir2_size - 1) * (dir1_size - 1) + skip_horizontal_edges;
1034  ++per_edge_idx;
1035  }
1036  skip_horizontal_edges = (dir2_size - 1) * (dir1_size - 1);
1037  // Reverse order loop for anti-clockwise edges order
1038  // To loop in reverse order with an unsigned type, we need to use a
1039  // trick. See: https://stackoverflow.com/a/5458251
1040  for (MeshIndex i_dir1 = dir1_size - 2; i_dir1 != static_cast<MeshIndex>(-1);
1041  --i_dir1) {
1042  perimeter_edges[per_edge_idx] = i_dir1 + skip_horizontal_edges;
1043  ++per_edge_idx;
1044  }
1045  skip_horizontal_edges = (dir2_size) * (dir1_size - 1);
1046  // Reverse order loop for anti-clockwise edges order
1047  // To loop in reverse order with an unsigned type, we need to use a
1048  // trick. See: https://stackoverflow.com/a/5458251
1049  for (MeshIndex i_dir2 = dir2_size - 2; i_dir2 != static_cast<MeshIndex>(-1);
1050  --i_dir2) {
1051  perimeter_edges[per_edge_idx] = i_dir2 + skip_horizontal_edges;
1052  ++per_edge_idx;
1053  }
1054 
1055  // 5. Create the faces edges
1056  const MeshIndex num_faces = (dir1_size - 1) * (dir2_size - 1);
1057  FaceEdges faces_edges(num_faces, EdgesIdsList(4));
1058  MeshIndex face_idx = 0;
1059  skip_horizontal_edges = (dir2_size) * (dir1_size - 1);
1060  for (MeshIndex i_dir2 = 0; i_dir2 < dir2_size - 1; ++i_dir2) {
1061  for (MeshIndex i_dir1 = 0; i_dir1 < dir1_size - 1; ++i_dir1) {
1062  faces_edges[face_idx][0] = i_dir1 + (dir1_size - 1) * i_dir2;
1063  faces_edges[face_idx][1] =
1064  i_dir2 + (dir2_size - 1) * (i_dir1 + 1) + skip_horizontal_edges;
1065  faces_edges[face_idx][2] = i_dir1 + (dir1_size - 1) * (i_dir2 + 1);
1066  faces_edges[face_idx][3] =
1067  i_dir2 + (dir2_size - 1) * (i_dir1) + skip_horizontal_edges;
1068  ++face_idx;
1069  }
1070  }
1071 
1072  // 6. Create the trimesh
1073 
1074  TriMesh trimesh;
1075 
1076  trimesh.set_vertices(std::move(points));
1077  trimesh.set_edges(std::move(edges));
1078  trimesh.set_perimeter_edges(std::move(perimeter_edges));
1079  trimesh.set_faces_edges(std::move(faces_edges));
1080  trimesh.set_number_of_faces(num_faces * 2);
1081 
1082  return trimesh;
1083 }
