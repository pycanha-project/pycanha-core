301 void sort_triangles() {
302 // Create a vector of indices
303 // TODO: Change indices to Eigen object (for avoid castings)
304 std::vector<MeshIndex> indices(
305 static_cast<MeshIndex>(_triangles.rows()));
306
307 if (_triangles.rows() == 0) {
308 return;
309 }
310
311 std::iota(indices.begin(), indices.end(), 0);
312
313 // Sort the indices according to the face ids
314 std::sort(indices.begin(), indices.end(),
315 [this](MeshIndex i1, MeshIndex i2) {
316 return _face_ids[i1] < _face_ids[i2];
317 });
318
319 // Sort the triangles
320 TrianglesList sorted_triangles(_triangles.rows(), _triangles.cols());
321 for (VectorIndex i = 0; i < indices.size(); ++i) {
322 sorted_triangles.row(static_cast<Index>(i)) =
323 _triangles.row(indices[i]);
324 }
325 _triangles = std::move(sorted_triangles);
326
327 // Reorder the face ids
328 FaceIdsList sorted_face_ids(_face_ids.size());
329 for (Index i = 0; i < static_cast<Index>(indices.size()); ++i) {
330 sorted_face_ids[i] =
331 _face_ids[indices[static_cast<VectorIndex>(i)]];
332 // std::cout << sorted_face_ids[i] << "\n";
333 }
334
335 _face_ids = std::move(sorted_face_ids);
336 _are_triangles_sorted = true;
337 }

339 void compute_areas() {
340 // Cummulative areas only make sense if the triangles are sorted
341 if (!_are_triangles_sorted) {
342 sort_triangles();
343 }
344
345 // Size of triangles list
346 auto N = static_cast<MeshIndex>(_triangles.rows());
347
348 if (N == 0) {
349 return;
350 }
351
352 // Resize _cumareas to hold N elements
353 _cumareas.resize(N);
354
355 MeshIndex current_face = _face_ids[0]; // Should be 0
356 double cum_area = 0.0;
357 for (MeshIndex i = 0; i < N; ++i) {
358 // If we are on a new face, reset the cumulative area
359 if (_face_ids[i] != current_face) {
360 current_face = _face_ids[i];
361 cum_area = 0.0;
362 }
363
364 // Get vertices of triangle i
365 const Point3D p1 = _vertices.row(_triangles(i, 0));
366 const Point3D p2 = _vertices.row(_triangles(i, 1));
367 const Point3D p3 = _vertices.row(_triangles(i, 2));
368
369 // Compute area of triangle i
370 cum_area += 0.5 * ((p2 - p1).cross(p3 - p1)).norm();
371
372 // Store cumulative area
373 _cumareas[i] = static_cast<float>(cum_area);
374 }
375 }

975 // TODO(FEATURE)

1055 TriMesh get_geometry_mesh(MeshIndex mesh_idx) {
1056 // Create TriMesh object corresponding to the mesh of the geometry with
1057 // id mesh_idx Check that the mesh_idx is valid
1058 if (mesh_idx >= _n_geometries) {
1059 throw std::invalid_argument("Invalid mesh index");
1060 }
1061
1062 // Get the start and end indices of the vertices, triangles and edges
1063 const MeshIndex start_vertices = _geometries_vertices[mesh_idx];
1064 const MeshIndex end_vertices = _geometries_vertices[mesh_idx + 1];
1065 const MeshIndex start_triangles = _geometries_triangles[mesh_idx];
1066 const MeshIndex end_triangles = _geometries_triangles[mesh_idx + 1];
1067 const MeshIndex start_edges = _geometries_edges[mesh_idx];
1068 const MeshIndex end_edges = _geometries_edges[mesh_idx + 1];
1069 const MeshIndex start_perimeter_edges =
1070 _geometries_perimeter_edges[mesh_idx];
1071 const MeshIndex end_perimeter_edges =
1072 _geometries_perimeter_edges[mesh_idx + 1];
1073
1074 // Copy the content to new objects
1075 // Vertices
1076 VerticesListDouble new_vertices =
1077 _vertices.block(start_vertices, 0, end_vertices - start_vertices, 3)
1078 .cast<double>();
1079 // Triangles
1080 TrianglesList new_triangles = _triangles.block(
1081 start_triangles, 0, end_triangles - start_triangles, 3);
1082 new_triangles.array() -= start_vertices;
1083 // Faces id
1084 FaceIdsList new_face_ids = _face_ids.block(
1085 start_triangles, 0, end_triangles - start_triangles, 1);
1086
1087 // Edges
1088 EdgesList new_edges(_edges.begin() + start_edges,
1089 _edges.begin() + end_edges);
1090 for (auto& edge : new_edges) {
1091 // for index: edge) { index -= start_vertices; }
1092 std::transform(edge.begin(), edge.end(), edge.begin(),
1093 [start_vertices](auto index) {
1094 return index - start_vertices;
1095 });
1096 }
1097
1098 // Perimeter edges
1099 auto new_number_of_perimeter_edges =
1100 end_perimeter_edges - start_perimeter_edges;
1101 Edges new_perimeter_edges(new_number_of_perimeter_edges);
1102 // Use Eigen's block operation to select the subrange and then apply a
1103 // unary operation
1104 new_perimeter_edges =
1105 _perimeter_edges
1106 .segment(start_perimeter_edges, new_number_of_perimeter_edges)
1107 .unaryExpr(
1108 [start_edges](MeshIndex n) { return n - start_edges; });
1109
1110 // Faces edges
1111 // TODO(PERFORMANCE): Not needed if faces are sorted

1119

1121 for (auto& face_edges : new_faces_edges) {

1124 face_edges.begin(), face_edges.end(), face_edges.begin(),
1125 [start_edges](auto edge) { return edge - start_edges; });
1126 }
1127

1130 const ColorRGB surf_2_color = _back_colors[mesh_idx];
1131
1132 // Create the TriMesh object

1141 geometry_mesh.set_surface2_color(surf_2_color);

1244inline TriMesh create_2d_rectangular_mesh(const Eigen::VectorXd& dir1_mesh,
1245 const Eigen::VectorXd& dir2_mesh,
1246 double max_distance_points_dir1,
1247 double max_distance_points_dir2) {
1248 // Assert preconditions (only in debug mode)
1249 assert(dir1_mesh.size() >= 2);
1250 assert(dir2_mesh.size() >= 2);
1251 assert(2 * dir1_mesh.size() * dir2_mesh.size() <=
1252 std::numeric_limits<MeshIndex>::max());
1253 assert(is_sorted(dir1_mesh));
1254 assert(is_sorted(dir2_mesh));
1255
1256 // TODO(PERFORMANCE): Use Eigen instead of std::vector to avoid casting
1257
1258 // 1. Determine the number of points to reserve space
1259 auto dir1_size = static_cast<MeshIndex>(dir1_mesh.size());
1260 auto dir2_size = static_cast<MeshIndex>(dir2_mesh.size());
1261
1262 MeshIndex num_points_dir1 = dir1_size; // Minimum number of points
1263 MeshIndex num_points_dir2 = dir2_size; // Minimum number of points
1264
1265 std::vector<MeshIndex> additional_points_dir1(dir1_size - 1, 0);
1266 std::vector<MeshIndex> additional_points_dir2(dir2_size - 1, 0);
1267
1268 // Now check if additional points are needed based on
1269 // max_distance_points If max_distance is negative or zero, then we
1270 // don't add additional points
1271 if (max_distance_points_dir1 > LENGTH_TOL) {
1272 for (MeshIndex i = 0; i < dir1_size - 1; ++i) {
1273 const double distance = dir1_mesh[i + 1] - dir1_mesh[i];
1274 const auto num_additional_points = static_cast<MeshIndex>(
1275 std::floor(distance / max_distance_points_dir1));
1276 num_points_dir1 += num_additional_points;
1277 additional_points_dir1[i] = num_additional_points;
1278 }
1279 }
1280
1281 if (max_distance_points_dir2 > LENGTH_TOL) {
1282 for (MeshIndex i = 0; i < dir2_size - 1; ++i) {
1283 const double distance = dir2_mesh[i + 1] - dir2_mesh[i];
1284 const auto num_additional_points = static_cast<MeshIndex>(
1285 std::floor(distance / max_distance_points_dir2));
1286 num_points_dir2 += num_additional_points;
1287 additional_points_dir2[i] = num_additional_points;
1288 }
1289 }
1290
1291 // 2. Create the mesh points (3D points with z = 0)
1292 VerticesList points(num_points_dir1 * num_points_dir2, 3);
1293 std::vector<double> full_dir1_mesh(
1294 static_cast<VectorIndex>(num_points_dir1));
1295 std::vector<double> full_dir2_mesh(
1296 static_cast<VectorIndex>(num_points_dir2));
1297
1298 // Fill the full mesh arrays with the original and additional mesh
1299 // points
1300 MeshIndex p_idx = 0;
1301 for (MeshIndex i_dir1 = 0; i_dir1 < dir1_size - 1; ++i_dir1) {
1302 const double distance = (dir1_mesh[i_dir1 + 1] - dir1_mesh[i_dir1]) /
1303 (additional_points_dir1[i_dir1] + 1);
1304 for (MeshIndex i_add1 = 0; i_add1 <= additional_points_dir1[i_dir1];
1305 ++i_add1) {
1306 full_dir1_mesh[p_idx] =
1307 dir1_mesh[i_dir1] + static_cast<double>(i_add1) * distance;
1308 ++p_idx;
1309 }
1310 }
1311 full_dir1_mesh[full_dir1_mesh.size() - 1] = dir1_mesh[dir1_size - 1];
1312 p_idx = 0;
1313 for (MeshIndex i_dir2 = 0; i_dir2 < dir2_size - 1; ++i_dir2) {
1314 const double distance = (dir2_mesh[i_dir2 + 1] - dir2_mesh[i_dir2]) /
1315 (additional_points_dir2[i_dir2] + 1);
1316 for (MeshIndex i_add1 = 0; i_add1 <= additional_points_dir2[i_dir2];
1317 ++i_add1) {
1318 full_dir2_mesh[p_idx] =
1319 dir2_mesh[i_dir2] + static_cast<double>(i_add1) * distance;
1320 ++p_idx;
1321 }
1322 }
1323 full_dir2_mesh[full_dir2_mesh.size() - 1] = dir2_mesh[dir2_size - 1];
1324
1325 // Fill the points array
1326 p_idx = 0;
1327 for (MeshIndex i_dir2 = 0; i_dir2 < num_points_dir2; ++i_dir2) {
1328 for (MeshIndex i_dir1 = 0; i_dir1 < num_points_dir1; ++i_dir1) {
1329 points(p_idx, 0) = full_dir1_mesh[static_cast<VectorIndex>(i_dir1)];
1330 points(p_idx, 1) = full_dir2_mesh[static_cast<VectorIndex>(i_dir2)];
1331 points(p_idx, 2) = 0.0;
1332 ++p_idx;
1333 }
1334 }
1335
1336 // 3. Create the edges
1337
1338 // Reserve space for the edges
1339 EdgesList edges((dir1_size - 1) * (dir2_size) +
1340 (dir1_size) * (dir2_size - 1));
1341
1342 // Fill the edges
1343
1344 // First the edges in direction 1
1345 MeshIndex e_idx = 0;
1346 MeshIndex skip_dir2_rows = 0;
1347 for (MeshIndex i_dir2 = 0; i_dir2 < dir2_size; ++i_dir2) {
1348 p_idx = num_points_dir1 * (i_dir2 + skip_dir2_rows);
1349 for (MeshIndex i_dir1 = 0; i_dir1 < dir1_size - 1; ++i_dir1) {
1350 const MeshIndex num_edge_points =
1351 additional_points_dir1[i_dir1] + 2;
1352 Edges edge(num_edge_points);
1353#if defined(__GNUC__)
1354#pragma GCC diagnostic push
1355#pragma GCC diagnostic ignored "-Wnull-dereference"
1356#endif
1357 // num_edge_points where a compiled constant, it would be fine
1358 edge[0] = p_idx;
1359#if defined(__GNUC__)
1360#pragma GCC diagnostic pop
1361#endif
1362
1363 for (MeshIndex i_add1 = 0; i_add1 <= additional_points_dir1[i_dir1];
1364 ++i_add1) {
1365 edge[i_add1 + 1] = p_idx + i_add1 + 1;
1366 }
1367 edges[e_idx] = edge;
1368 e_idx++;
1369 p_idx += num_edge_points - 1;
1370 }
1371 if (i_dir2 < dir2_size - 1) {
1372 skip_dir2_rows += additional_points_dir2[i_dir2];
1373 }
1374 }
1375
1376 // Then the edges in direction 2
1377 MeshIndex skip_dir1_cols = 0;
1378 for (MeshIndex i_dir1 = 0; i_dir1 < dir1_size; ++i_dir1) {
1379 p_idx = i_dir1 + skip_dir1_cols;
1380 for (MeshIndex i_dir2 = 0; i_dir2 < dir2_size - 1; ++i_dir2) {
1381 const MeshIndex num_edge_points =
1382 additional_points_dir2[i_dir2] + 2;
1383 Edges edge(num_edge_points);
1384#if defined(__GNUC__)
1385#pragma GCC diagnostic push
1386#pragma GCC diagnostic ignored "-Wnull-dereference"
1387#endif
1388 // num_edge_points where a compiled constant, it would be fine
1389 edge[0] = p_idx;
1390#if defined(__GNUC__)
1391#pragma GCC diagnostic pop
1392#endif
1393 for (MeshIndex i_add2 = 0; i_add2 <= additional_points_dir2[i_dir2];
1394 ++i_add2) {
1395 edge[i_add2 + 1] = p_idx + (i_add2 + 1) * num_points_dir1;
1396 }
1397 edges[e_idx] = edge;
1398 e_idx++;
1399 p_idx += (num_edge_points - 1) * num_points_dir1;
1400 }
1401 if (i_dir1 < dir1_size - 1) {
1402 skip_dir1_cols += additional_points_dir1[i_dir1];
1403 }
1404 }
1405
1406 // 4. Create the permiter edges. Anti-clockwise order starting from P01
1407 // -> dir1 edge
1408 const MeshIndex num_perimeter_edges = 2 * (dir1_size + dir2_size - 2);
1409 EdgesIdsList perimeter_edges(num_perimeter_edges);
1410 MeshIndex per_edge_idx = 0;
1411 for (MeshIndex i_dir1 = 0; i_dir1 < dir1_size - 1; ++i_dir1) {
1412 perimeter_edges[per_edge_idx] = i_dir1;
1413 ++per_edge_idx;
1414 }
1415 MeshIndex skip_horizontal_edges = (dir2_size) * (dir1_size - 1);
1416 for (MeshIndex i_dir2 = 0; i_dir2 < dir2_size - 1; ++i_dir2) {
1417 perimeter_edges[per_edge_idx] =
1418 i_dir2 + (dir2_size - 1) * (dir1_size - 1) + skip_horizontal_edges;
1419 ++per_edge_idx;
1420 }
1421 skip_horizontal_edges = (dir2_size - 1) * (dir1_size - 1);
1422 // Reverse order loop for anti-clockwise edges order
1423 // To loop in reverse order with an unsigned type, we need to use a
1424 // trick. See: https://stackoverflow.com/a/5458251
1425 for (MeshIndex i_dir1 = dir1_size - 2; i_dir1 != static_cast<MeshIndex>(-1);
1426 --i_dir1) {
1427 perimeter_edges[per_edge_idx] = i_dir1 + skip_horizontal_edges;
1428 ++per_edge_idx;
1429 }
1430 skip_horizontal_edges = (dir2_size) * (dir1_size - 1);
1431 // Reverse order loop for anti-clockwise edges order
1432 // To loop in reverse order with an unsigned type, we need to use a
1433 // trick. See: https://stackoverflow.com/a/5458251
1434 for (MeshIndex i_dir2 = dir2_size - 2; i_dir2 != static_cast<MeshIndex>(-1);
1435 --i_dir2) {
1436 perimeter_edges[per_edge_idx] = i_dir2 + skip_horizontal_edges;
1437 ++per_edge_idx;
1438 }
1439
1440 // 5. Create the faces edges
1441 const MeshIndex num_faces = (dir1_size - 1) * (dir2_size - 1);
1442 FaceEdges faces_edges(num_faces, EdgesIdsList(4));
1443 MeshIndex face_idx = 0;
1444 skip_horizontal_edges = (dir2_size) * (dir1_size - 1);
1445 for (MeshIndex i_dir2 = 0; i_dir2 < dir2_size - 1; ++i_dir2) {
1446 for (MeshIndex i_dir1 = 0; i_dir1 < dir1_size - 1; ++i_dir1) {
1447 faces_edges[face_idx][0] = i_dir1 + (dir1_size - 1) * i_dir2;
1448 faces_edges[face_idx][1] =
1449 i_dir2 + (dir2_size - 1) * (i_dir1 + 1) + skip_horizontal_edges;
1450 faces_edges[face_idx][2] = i_dir1 + (dir1_size - 1) * (i_dir2 + 1);
1451 faces_edges[face_idx][3] =
1452 i_dir2 + (dir2_size - 1) * (i_dir1) + skip_horizontal_edges;
1453 ++face_idx;
1454 }
1455 }
1456
1457 // 6. Create the trimesh
1458
1459 TriMesh trimesh;
1460
1461 trimesh.set_vertices(std::move(points));
1462 trimesh.set_edges(std::move(edges));
1463 trimesh.set_perimeter_edges(std::move(perimeter_edges));
1464 trimesh.set_faces_edges(std::move(faces_edges));
1465 trimesh.set_number_of_faces(num_faces * 2);
1466
1467 return trimesh;
1468}

1477inline TriMesh create_2d_quadrilateral_mesh(
1478 const Eigen::VectorXd& dir1_mesh_normalized,
1479 const Eigen::VectorXd& dir2_mesh_normalized, const Point2D& p1,
1480 const Point2D& p2, const Point2D& p3, const Point2D& p4,
1481 double max_distance_points_hdir, double max_distance_points_vdir) {
1482 // Assert preconditions (only in debug mode)
1483 assert(dir1_mesh_normalized.size() >= 2);
1484 assert(dir2_mesh_normalized.size() >= 2);
1485 assert(2 * dir1_mesh_normalized.size() * dir2_mesh_normalized.size() <=
1486 std::numeric_limits<MeshIndex>::max());
1487 assert(is_sorted(dir1_mesh_normalized));
1488 assert(is_sorted(dir2_mesh_normalized));
1489
1490 // Normalize the mesh
1491 // Eigen::VectorXd dir1_mesh_normalized = dir1_mesh;
1492 // Eigen::VectorXd dir2_mesh_normalized = dir2_mesh;
1493 // Eigen::VectorXd dir1_mesh_normalized =
1494 // dir1_mesh / (dir1_mesh[dir1_mesh.size() - 1]);
1495 // Eigen::VectorXd dir2_mesh_normalized =
1496 // dir2_mesh / (dir2_mesh[dir2_mesh.size() - 1]);
1497 auto v12 = p2 - p1;
1498 auto v23 = p3 - p2;
1499 auto v14 = p4 - p1;
1500 auto v43 = p3 - p4;
1501
1502 const Eigen::VectorXd dir1_mesh = dir1_mesh_normalized * v12.norm();
1503 const Eigen::VectorXd dir2_mesh = dir2_mesh_normalized * v23.norm();
1504
1505 // 1. Determine the number of points to reserve space
1506 const auto dir1_size = static_cast<MeshIndex>(dir1_mesh.size());
1507 const auto dir2_size = static_cast<MeshIndex>(dir2_mesh.size());
1508
1509 MeshIndex num_points_edges =
1510 dir1_size * dir2_size; // Minimum number of points
1511
1512 // In the quadrilateral case, each line of the mesh has a different length,
1513 // so it might result in different numbers of points to be added. Note also
1514 // that the direction of the lines for dir1 and for dir2 changes from one
1515 // line to the next one.
1516 std::vector<std::vector<MeshIndex>> additional_points_dir1(
1517 dir2_size, std::vector<MeshIndex>(dir1_size - 1, 0));
1518 std::vector<std::vector<MeshIndex>> additional_points_dir2(
1519 dir1_size, std::vector<MeshIndex>(dir2_size - 1, 0));
1520
1521 // We also create a vector of vectors to store the dir_mesh of each line
1522 // dir1_meshes and dir2_meshes will contain the non-normalized mesh of each
1523 // line. To recover the actual points from the meshes, you would need the
1524 // start point and a direction. For dir1 and dir2 lines, the direction
1525 // changes from one line to the next one.
1526 std::vector<Eigen::VectorXd> dir1_meshes(dir2_size);
1527 std::vector<Eigen::VectorXd> dir2_meshes(dir1_size);
1528 std::vector<Vector2D> dir1_directions(dir2_size);
1529 std::vector<Vector2D> dir2_directions(dir1_size);
1530 std::vector<Point2D> dir1_start_points(dir2_size);
1531 std::vector<Point2D> dir2_start_points(dir1_size);
1532 for (MeshIndex i = 0; i < dir2_size; ++i) {
1533 auto start_point = p1 + v14 * dir2_mesh_normalized[i];
1534 auto end_point = p2 + v23 * dir2_mesh_normalized[i];
1535 dir1_start_points[i] = start_point;
1536 dir1_directions[i] = (end_point - start_point).normalized();
1537 const double distance = (end_point - start_point).norm();
1538 dir1_meshes[i] = dir1_mesh_normalized * distance;
1539 }
1540 for (MeshIndex i = 0; i < dir1_size; ++i) {
1541 auto start_point = p1 + v12 * dir1_mesh_normalized[i];
1542 auto end_point = p4 + v43 * dir1_mesh_normalized[i];
1543 dir2_start_points[i] = start_point;
1544 dir2_directions[i] = (end_point - start_point).normalized();
1545 const double distance = (end_point - start_point).norm();
1546 dir2_meshes[i] = dir2_mesh_normalized * distance;
1547 }
1548
1549 // Now check if additional points are needed based on
1550 // max_distance_points If max_distance is negative or zero, then we
1551 // don't add additional points
1552 if (max_distance_points_hdir > LENGTH_TOL) {
1553 // We loop over all the dir1 ("horizontal") lines
1554 for (MeshIndex j = 0; j < dir2_size; ++j) {
1555 for (MeshIndex i = 0; i < dir1_size - 1; ++i) {
1556 const Eigen::VectorXd& mesh =
1557 dir1_meshes[j]; // Mesh of the current line
1558 double hdistance = mesh[i + 1] - mesh[i];
1559 // Project the distance in the horizontal direction
1560 hdistance = hdistance * dir1_directions[j].norm();
1561 const auto num_additional_points = static_cast<MeshIndex>(
1562 std::floor(hdistance / max_distance_points_hdir));
1563 num_points_edges += num_additional_points;
1564 additional_points_dir1[j][i] = num_additional_points;
1565 }
1566 }
1567 }
1568
1569 if (max_distance_points_vdir > LENGTH_TOL) {
1570 // We loop over all the dir2 ("vertical") lines
1571 for (MeshIndex j = 0; j < dir1_size; ++j) {
1572 for (MeshIndex i = 0; i < dir2_size - 1; ++i) {
1573 const Eigen::VectorXd& mesh =
1574 dir2_meshes[j]; // Mesh of the current line
1575 double vdistance = mesh[i + 1] - mesh[i];
1576 // Project the distance in the vertical direction
1577 vdistance = vdistance * dir2_directions[j].norm();
1578 const auto num_additional_points = static_cast<MeshIndex>(
1579 std::floor(vdistance / max_distance_points_vdir));
1580 num_points_edges += num_additional_points;
1581 additional_points_dir2[j][i] = num_additional_points;
1582 }
1583 }
1584 }
1585
1586 // 2. Create the mesh points (3D points with z = 0)
1587
1588 // Fill the full mesh arrays with the original and additional mesh
1589 // points
1590 for (MeshIndex j = 0; j < dir2_size; ++j) {
1591 const Eigen::VectorXd& dir_mesh =
1592 dir1_meshes[j]; // Mesh of the current line
1593
1594 constexpr MeshIndex init_value = 0;
1595 const MeshIndex additional_points =
1596 std::accumulate(additional_points_dir1[j].begin(),
1597 additional_points_dir1[j].end(), init_value);
1598 Eigen::VectorXd full_dir_mesh(dir1_size + additional_points);
1599
1600 int idx = 0;
1601 for (MeshIndex i = 0; i < dir1_size - 1; ++i) {
1602 const double distance = (dir_mesh[i + 1] - dir_mesh[i]) /
1603 (additional_points_dir1[j][i] + 1);
1604 for (MeshIndex i_add = 0; i_add <= additional_points_dir1[j][i];
1605 ++i_add) {
1606 full_dir_mesh[idx] = dir_mesh[i] + distance * i_add;
1607 ++idx;
1608 }
1609 }
1610 full_dir_mesh[full_dir_mesh.size() - 1] = dir_mesh[dir1_size - 1];
1611 dir1_meshes[j] = std::move(full_dir_mesh);
1612 }
1613
1614 for (MeshIndex j = 0; j < dir1_size; ++j) {
1615 const Eigen::VectorXd& dir_mesh =
1616 dir2_meshes[j]; // Mesh of the current line
1617
1618 constexpr MeshIndex init_value = 0;
1619 const MeshIndex additional_points =
1620 std::accumulate(additional_points_dir2[j].begin(),
1621 additional_points_dir2[j].end(), init_value);
1622 Eigen::VectorXd full_dir_mesh(dir2_size + additional_points);
1623
1624 int idx = 0;
1625 for (MeshIndex i = 0; i < dir2_size - 1; ++i) {
1626 const double distance = (dir_mesh[i + 1] - dir_mesh[i]) /
1627 (additional_points_dir2[j][i] + 1);
1628 for (MeshIndex i_add = 0; i_add <= additional_points_dir2[j][i];
1629 ++i_add) {
1630 full_dir_mesh[idx] = dir_mesh[i] + distance * i_add;
1631 ++idx;
1632 }
1633 }
1634 full_dir_mesh[full_dir_mesh.size() - 1] = dir_mesh[dir2_size - 1];
1635 dir2_meshes[j] = std::move(full_dir_mesh);
1636 }
1637
1638 // Loop over the faces to determine the number of interior points to be
1639 // added
1640 Eigen::Matrix<MeshIndex, Eigen::Dynamic, Eigen::Dynamic>
1641 additional_points_faces_dir1((dir2_size - 1), (dir1_size - 1));
1642 Eigen::Matrix<MeshIndex, Eigen::Dynamic, Eigen::Dynamic>
1643 additional_points_faces_dir2((dir2_size - 1), (dir1_size - 1));
1644
1645 MeshIndex num_interior_points = 0;
1646 for (MeshIndex j = 0; j < dir2_size - 1; ++j) {
1647 for (MeshIndex i = 0; i < dir1_size - 1; ++i) {
1648 additional_points_faces_dir1(j, i) = std::max(
1649 additional_points_dir1[j][i], additional_points_dir1[j + 1][i]);
1650 additional_points_faces_dir2(j, i) = std::max(
1651 additional_points_dir2[i][j], additional_points_dir2[i + 1][j]);
1652 num_interior_points += additional_points_faces_dir1(j, i) *
1653 additional_points_faces_dir2(j, i);
1654 }
1655 }
1656
1657 // Fill the points array
1658
1659 VerticesList points(num_points_edges + num_interior_points, 3);
1660 MeshIndex p_idx = 0;
1661
1662 // Loop over dir1 lines
1663 for (MeshIndex j = 0; j < dir2_size; ++j) {
1664 const Eigen::VectorXd& dir_mesh =
1665 dir1_meshes[j]; // Mesh of the current line
1666 const Vector2D& dir = dir1_directions[j];
1667 const Point2D& start_point = dir1_start_points[j];
1668 for (const auto& mesh_i : dir_mesh) {
1669 points(p_idx, 0) = start_point.x() + mesh_i * dir.x();
1670 points(p_idx, 1) = start_point.y() + mesh_i * dir.y();
1671 points(p_idx, 2) = 0.0;
1672 ++p_idx;
1673 }
1674 }
1675
1676 // Loop over dir2 lines
1677 for (MeshIndex j = 0; j < dir1_size; ++j) {
1678 const Eigen::VectorXd& dir_mesh =
1679 dir2_meshes[j]; // Mesh of the current line
1680 const Vector2D& dir = dir2_directions[j];
1681 const Point2D& start_point = dir2_start_points[j];
1682 int line_point_idx = 0;
1683 for (MeshIndex i = 0; i < dir2_mesh_normalized.size() - 1; ++i) {
1684 // We skip the non-additional points as those are already added
1685 for (MeshIndex iadd = 0; iadd < additional_points_dir2[j][i];
1686 ++iadd) {
1687 points(p_idx, 0) =
1688 start_point.x() + dir_mesh[line_point_idx + 1] * dir.x();
1689 points(p_idx, 1) =
1690 start_point.y() + dir_mesh[line_point_idx + 1] * dir.y();
1691 points(p_idx, 2) = 0.0;
1692 ++p_idx;
1693 ++line_point_idx;
1694 }
1695 ++line_point_idx;
1696 }
1697 }
1698
1699 // 3. Create the edges
1700
1701 // Reserve space for the edges
1702 EdgesList edges((dir1_size - 1) * (dir2_size) +
1703 (dir1_size) * (dir2_size - 1));
1704
1705 // Fill the edges
1706
1707 // First the edges in direction 1
1708 MeshIndex e_idx = 0;
1709 p_idx = 0;
1710
1711 // Loop over dir1 lines (edges in dir1)
1712 for (MeshIndex j = 0; j < dir2_size; ++j) {
1713 for (MeshIndex i = 0; i < dir1_size - 1; ++i) {
1714 const MeshIndex num_edge_points = additional_points_dir1[j][i] + 2;
1715 Edges edge(num_edge_points);
1716#if defined(__GNUC__)
1717#pragma GCC diagnostic push
1718#pragma GCC diagnostic ignored "-Wnull-dereference"
1719#endif
1720 // num_edge_points where a compiled constant, it would be fine
1721 edge[0] = p_idx;
1722#if defined(__GNUC__)
1723#pragma GCC diagnostic pop
1724#endif
1725 for (MeshIndex i_add = 0; i_add <= additional_points_dir1[j][i];
1726 ++i_add) {
1727 edge[i_add + 1] = p_idx + i_add + 1;
1728 }
1729 edges[e_idx] = edge;
1730 e_idx++;
1731 p_idx += num_edge_points - 1;
1732 }
1733 p_idx++;
1734 }
1735
1736 // Loop over dir2 lines (edges in dir2)
1737 for (MeshIndex j = 0; j < dir1_size - 1; ++j) {
1738 for (MeshIndex i = 0; i < dir2_size - 1; ++i) {
1739 const MeshIndex lower_edge_idx = (dir1_size - 1) * i + j;
1740 const MeshIndex upper_edge_idx = (dir1_size - 1) * (i + 1) + j;
1741 const MeshIndex num_edge_points = additional_points_dir2[j][i] + 2;
1742 Edges edge(num_edge_points);
1743#if defined(__GNUC__)
1744#pragma GCC diagnostic push
1745#pragma GCC diagnostic ignored "-Wnull-dereference"
1746#endif
1747 // num_edge_points where a compiled constant, it would be fine
1748 edge[0] = edges[lower_edge_idx][0];
1749#if defined(__GNUC__)
1750#pragma GCC diagnostic pop
1751#endif
1752 edge[num_edge_points - 1] = edges[upper_edge_idx][0];
1753 for (MeshIndex i_add = 0; i_add < additional_points_dir2[j][i];
1754 ++i_add) {
1755 edge[i_add + 1] = p_idx;
1756 p_idx++;
1757 }
1758 edges[e_idx] = edge;
1759 e_idx++;
1760 }
1761 }
1762 // Last edge
1763 for (MeshIndex i = 0; i < dir2_size - 1; ++i) {
1764 const MeshIndex lower_edge_idx = (dir1_size - 1) * i + dir1_size - 2;
1765 const MeshIndex upper_edge_idx =
1766 (dir1_size - 1) * (i + 1) + dir1_size - 2;
1767 const MeshIndex num_edge_points =
1768 additional_points_dir2[dir1_size - 1][i] + 2;
1769 Edges edge(num_edge_points);
1770#if defined(__GNUC__)
1771#pragma GCC diagnostic push
1772#pragma GCC diagnostic ignored "-Wnull-dereference"
1773#endif
1774 edge[0] = edges[lower_edge_idx][edges[lower_edge_idx].size() - 1];
1775#if defined(__GNUC__)
1776#pragma GCC diagnostic pop
1777#endif
1778 edge[num_edge_points - 1] =
1779 edges[upper_edge_idx][edges[upper_edge_idx].size() - 1];
1780 for (MeshIndex i_add = 0;
1781 i_add < additional_points_dir2[dir1_size - 1][i]; ++i_add) {
1782 edge[i_add + 1] = p_idx;
1783 p_idx++;
1784 }
1785 edges[e_idx] = edge;
1786 e_idx++;
1787 }
1788
1789 // 4. Create the permiter edges. Anti-clockwise order starting from P01
1790 // -> dir1 edge
1791 const MeshIndex num_perimeter_edges = 2 * (dir1_size + dir2_size - 2);
1792 EdgesIdsList perimeter_edges(num_perimeter_edges);
1793 MeshIndex per_edge_idx = 0;
1794 for (MeshIndex i_dir1 = 0; i_dir1 < dir1_size - 1; ++i_dir1) {
1795 perimeter_edges[per_edge_idx] = i_dir1;
1796 ++per_edge_idx;
1797 }
1798 MeshIndex skip_horizontal_edges = (dir2_size) * (dir1_size - 1);
1799 for (MeshIndex i_dir2 = 0; i_dir2 < dir2_size - 1; ++i_dir2) {
1800 perimeter_edges[per_edge_idx] =
1801 i_dir2 + (dir2_size - 1) * (dir1_size - 1) + skip_horizontal_edges;
1802 ++per_edge_idx;
1803 }
1804 skip_horizontal_edges = (dir2_size - 1) * (dir1_size - 1);
1805 // Reverse order loop for anti-clockwise edges order
1806 // To loop in reverse order with an unsigned type, we need to use a
1807 // trick. See: https://stackoverflow.com/a/5458251
1808 for (MeshIndex i_dir1 = dir1_size - 2; i_dir1 != static_cast<MeshIndex>(-1);
1809 --i_dir1) {
1810 perimeter_edges[per_edge_idx] = i_dir1 + skip_horizontal_edges;
1811 ++per_edge_idx;
1812 }
1813
1814 skip_horizontal_edges = (dir2_size) * (dir1_size - 1);
1815 // Reverse order loop for anti-clockwise edges order
1816 // To loop in reverse order with an unsigned type, we need to use a
1817 // trick. See: https://stackoverflow.com/a/5458251
1818 for (MeshIndex i_dir2 = dir2_size - 2; i_dir2 != static_cast<MeshIndex>(-1);
1819 --i_dir2) {
1820 perimeter_edges[per_edge_idx] = i_dir2 + skip_horizontal_edges;
1821 ++per_edge_idx;
1822 }
1823
1824 // 5. Create the faces edges
1825 const MeshIndex num_faces = (dir1_size - 1) * (dir2_size - 1);
1826 FaceEdges faces_edges(num_faces, EdgesIdsList(4));
1827 MeshIndex face_idx = 0;
1828 skip_horizontal_edges = (dir2_size) * (dir1_size - 1);
1829 for (MeshIndex i_dir2 = 0; i_dir2 < dir2_size - 1; ++i_dir2) {
1830 for (MeshIndex i_dir1 = 0; i_dir1 < dir1_size - 1; ++i_dir1) {
1831 faces_edges[face_idx][0] = i_dir1 + (dir1_size - 1) * i_dir2;
1832 faces_edges[face_idx][1] =
1833 i_dir2 + (dir2_size - 1) * (i_dir1 + 1) + skip_horizontal_edges;
1834 faces_edges[face_idx][2] = i_dir1 + (dir1_size - 1) * (i_dir2 + 1);
1835 faces_edges[face_idx][3] =
1836 i_dir2 + (dir2_size - 1) * (i_dir1) + skip_horizontal_edges;
1837 ++face_idx;
1838 }
1839 }
1840
1841 // 2.5 Create the interior points
1842 // Loop over the faces adding the interior points (not used in the edges)
1843 if (num_interior_points > 0) {
1844 p_idx = num_points_edges;
1845 for (MeshIndex j = 0; j < dir2_size - 1; ++j) {
1846 for (MeshIndex i = 0; i < dir1_size - 1; ++i) {
1847 const MeshIndex i_face = j * (dir1_size - 1) + i;
1848 // Copying the points (instead of referencing) because points is
1849 // col-major
1850 const Point2D p1_f(
1851 points.row(edges[faces_edges[i_face][0]][0]).head<2>());
1852 const Point2D p2_f(
1853 points.row(edges[faces_edges[i_face][1]][0]).head<2>());
1854
1855 const Point2D p3_f(
1856 points.row(edges[faces_edges[i_face][2]].tail<1>().value())
1857 .head<2>());
1858 const Point2D p4_f(
1859 points.row(edges[faces_edges[i_face][3]].tail<1>().value())
1860 .head<2>());
1861
1862 const MeshIndex dir1_add_points =
1863 additional_points_faces_dir1(j, i);
1864 const MeshIndex dir2_add_points =
1865 additional_points_faces_dir2(j, i);
1866
1867 Vector2D v14_f = p4_f - p1_f;
1868 v14_f = v14_f / (dir2_add_points + 1);
1869 Vector2D v23_f = p3_f - p2_f;
1870 v23_f = v23_f / (dir2_add_points + 1);
1871
1872 for (MeshIndex ip_dir2 = 1; ip_dir2 <= dir2_add_points;
1873 ++ip_dir2) {
1874 auto start_point = p1_f + v14_f * (ip_dir2);
1875 auto end_point = p2_f + v23_f * (ip_dir2);
1876 Vector2D vdir1 = end_point - start_point;
1877 vdir1 = vdir1 / (dir1_add_points + 1);
1878 for (MeshIndex ip_dir1 = 1; ip_dir1 <= dir1_add_points;
1879 ++ip_dir1) {
1880 points(p_idx, 0) =
1881 start_point.x() + vdir1.x() * ip_dir1;
1882 points(p_idx, 1) =
1883 start_point.y() + vdir1.y() * ip_dir1;
1884 points(p_idx, 2) = 0.0;
1885 ++p_idx;
1886 }
1887 }
1888 }
1889 }
1890 }
1891
1892 // 6. Create the trimesh
1893
1894 TriMesh trimesh;
1895
1896 trimesh.set_vertices(std::move(points));
1897 trimesh.set_edges(std::move(edges));
1898 trimesh.set_perimeter_edges(std::move(perimeter_edges));
1899 trimesh.set_faces_edges(std::move(faces_edges));
1900 trimesh.set_number_of_faces(num_faces * 2);
1901
1902 return trimesh;
1903}
