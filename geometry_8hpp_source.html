236 void triangulate_post_processed_cutted_mesh() {
237 // Call the CDT mesher to resolve the triangulation
238 trimesher::cdt_trimesher(*_tri_mesh);
239
240 // Reconstruct the face edges
241 _primitive->reconstruct_face_edges_2d(_tri_mesh, *_thermal_mesh);
242
243 // Assign face ids
244 for (MeshIndex i = 0; i < _tri_mesh->get_triangles().rows(); i++) {
245 auto tri = _tri_mesh->get_triangles().row(i);
246 const Point2D p0_2d{_tri_mesh->get_vertices()(tri[0], 0),
247 _tri_mesh->get_vertices()(tri[0], 1)};
248 const Point2D p1_2d{_tri_mesh->get_vertices()(tri[1], 0),
249 _tri_mesh->get_vertices()(tri[1], 1)};
250 const Point2D p2_2d{_tri_mesh->get_vertices()(tri[2], 0),
251 _tri_mesh->get_vertices()(tri[2], 1)};
252
253 const Point2D centroid = (p0_2d + p1_2d + p2_2d) / 3.0;
254
255 // Get the face id
256 _tri_mesh->get_face_ids()[i] =
257 _primitive->get_faceid_from_uv(*_thermal_mesh, centroid);
258 }
259
260 // Transform 2D points to 3D
261 for (int i = 0; i < _tri_mesh->get_vertices().rows(); i++) {
262 const Point2D point2d{_tri_mesh->get_vertices()(i, 0),
263 _tri_mesh->get_vertices()(i, 1)};
264 _tri_mesh->get_vertices().row(i) =
265 _primitive->from_2d_to_3d(point2d);
266 }
267 // sort the mesh
268 _tri_mesh->sort_triangles();
269 // compute areas
270 _tri_mesh->compute_areas();
271 }

720inline void GeometryGroup::iterate_create_cut_groups(
721 const CoordinateTransformation& global_transformation,
722 std::vector<std::shared_ptr<GeometryMeshedItem>>&
723 cutted_geometry_meshed_items,
724 std::vector<std::vector<std::shared_ptr<Primitive>>>& cutting_primitives) {
725 auto index_geometry_below_this_group_starts =
726 cutted_geometry_meshed_items.size();
727
728 for (const auto& child_group : _geometry_groups) {
729 child_group->iterate_create_cut_groups(
730 global_transformation.chain(*(child_group->get_transformation())),
731 cutted_geometry_meshed_items, cutting_primitives);
732 }
733
734 for (const auto& child_group_cutted : _geometry_groups_cutted) {
735 child_group_cutted->iterate_create_cut_groups(
736 global_transformation.chain(
737 *(child_group_cutted->get_transformation())),
738 cutted_geometry_meshed_items, cutting_primitives);
739 }
740
741 for (const auto& item : _geometry_items) {
742 // Creates new GeometryMeshedItem
743 const std::string new_name = item->get_name() + "_cutted";
744 auto current_coord_transf =
745 global_transformation.chain(*(item->get_transformation()));
746 const PrimitivePtr new_primitive =
747 item->get_primitive()->transform(current_coord_transf);
748 const TransformationPtr new_coord_transf =
749 std::make_shared<CoordinateTransformation>();
750 auto th_mesh = item->get_thermal_mesh();
751
752 auto cutted_item = std::make_shared<GeometryMeshedItem>(
753 new_name, new_primitive, new_coord_transf, th_mesh);
754
755 // Add the new GeometryMeshedItem to the vector
756 cutted_geometry_meshed_items.push_back(cutted_item);
757
758 // This adds an empty vector<std::shared_ptr<Primitive>> directly.
759 cutting_primitives.emplace_back();
760 }
761
762 // Check if I'm actually a GeometryGroupCutted by trying to dynamic casting
763 // *this to GeometryGroupCutted
764 if (auto* group_cutted = dynamic_cast<GeometryGroupCutted*>(this)) {
765 std::vector<std::shared_ptr<Primitive>> this_group_cutting_primitives;
766 // If I'm a GeometryGroupCutted, iterate over all cutting primitives
767 for (const auto& item : group_cutted->get_cutting_geometry_items()) {
768 auto current_coord_transf =
769 global_transformation.chain(*(item->get_transformation()));
770
771 auto cutting_primitive =
772 item->get_primitive()->transform(current_coord_transf);
773 this_group_cutting_primitives.push_back(cutting_primitive);
774 }
775
776 // Insert the cutting primitives to all the items below and in this
777 // group
778 for (auto i = index_geometry_below_this_group_starts;
779 i < cutting_primitives.size(); ++i) {
780 cutting_primitives[i].insert(cutting_primitives[i].end(),
781 this_group_cutting_primitives.begin(),
782 this_group_cutting_primitives.end());
783 }
784 }
785}
