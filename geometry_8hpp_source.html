240  // Call the CDT mesher to resolve the triangulation
241  trimesher::cdt_trimesher(*_tri_mesh);
242 
243  // Reconstruct the face edges
244  _primitive->reconstruct_face_edges_2d(_tri_mesh, *_thermal_mesh);
245 
246  // Assign face ids
247  for (MeshIndex i = 0; i < _tri_mesh->get_triangles().rows(); i++) {
248  auto tri = _tri_mesh->get_triangles().row(i);
249  const Point2D p0_2d{_tri_mesh->get_vertices()(tri[0], 0),
250  _tri_mesh->get_vertices()(tri[0], 1)};
251  const Point2D p1_2d{_tri_mesh->get_vertices()(tri[1], 0),
252  _tri_mesh->get_vertices()(tri[1], 1)};
253  const Point2D p2_2d{_tri_mesh->get_vertices()(tri[2], 0),
254  _tri_mesh->get_vertices()(tri[2], 1)};
255 
256  const Point2D centroid = (p0_2d + p1_2d + p2_2d) / 3.0;
257 
258  // Get the face id
259  _tri_mesh->get_face_ids()[i] =
260  _primitive->get_faceid_from_uv(*_thermal_mesh, centroid);
261  }
262 
263  // Transform 2D points to 3D
264  for (int i = 0; i < _tri_mesh->get_vertices().rows(); i++) {
265  const Point2D point2d{_tri_mesh->get_vertices()(i, 0),
266  _tri_mesh->get_vertices()(i, 1)};
267  _tri_mesh->get_vertices().row(i) =
268  _primitive->from_2d_to_3d(point2d);
269  }
270  // sort the mesh
271  _tri_mesh->sort_triangles();
272  // compute areas
273  _tri_mesh->compute_areas();
274  }

723 inline void GeometryGroup::iterate_create_cut_groups(
724  const CoordinateTransformation& global_transformation,
725  std::vector<std::shared_ptr<GeometryMeshedItem>>&
726  cutted_geometry_meshed_items,
727  std::vector<std::vector<std::shared_ptr<Primitive>>>& cutting_primitives) {
728  auto index_geometry_below_this_group_starts =
729  cutted_geometry_meshed_items.size();
730 
731  for (const auto& child_group : _geometry_groups) {
732  child_group->iterate_create_cut_groups(
733  global_transformation.chain(*(child_group->get_transformation())),
734  cutted_geometry_meshed_items, cutting_primitives);
735  }
736 
737  for (const auto& child_group_cutted : _geometry_groups_cutted) {
738  child_group_cutted->iterate_create_cut_groups(
739  global_transformation.chain(
740  *(child_group_cutted->get_transformation())),
741  cutted_geometry_meshed_items, cutting_primitives);
742  }
743 
744  for (const auto& item : _geometry_items) {
745  // Creates new GeometryMeshedItem
746  const std::string new_name = item->get_name() + "_cutted";
747  auto current_coord_transf =
748  global_transformation.chain(*(item->get_transformation()));
749  const PrimitivePtr new_primitive =
750  item->get_primitive()->transform(current_coord_transf);
751  const TransformationPtr new_coord_transf =
752  std::make_shared<CoordinateTransformation>();
753  auto th_mesh = item->get_thermal_mesh();
754 
755  auto cutted_item = std::make_shared<GeometryMeshedItem>(
756  new_name, new_primitive, new_coord_transf, th_mesh);
757 
758  // Add the new GeometryMeshedItem to the vector
759  cutted_geometry_meshed_items.push_back(cutted_item);
760 
761  // This adds an empty vector<std::shared_ptr<Primitive>> directly.
762  cutting_primitives.emplace_back();
763  }
764 
765  std::vector<std::shared_ptr<Primitive>> this_group_cutting_primitives;
766  // Check if I'm actually a GeometryGroupCutted by trying to dynamic casting
767  // *this to GeometryGroupCutted
768  if (auto* group_cutted = dynamic_cast<GeometryGroupCutted*>(this)) {
769  // If I'm a GeometryGroupCutted, iterate over all cutting primitives
770  for (const auto& item : group_cutted->get_cutting_geometry_items()) {
771  auto current_coord_transf =
772  global_transformation.chain(*(item->get_transformation()));
773 
774  auto cutting_primitive =
775  item->get_primitive()->transform(current_coord_transf);
776  this_group_cutting_primitives.push_back(cutting_primitive);
777  }
778 
779  // Insert the cutting primitives to all the items below and in this
780  // group
781  for (auto i = index_geometry_below_this_group_starts;
782  i < cutting_primitives.size(); ++i) {
783  cutting_primitives[i].insert(cutting_primitives[i].end(),
784  this_group_cutting_primitives.begin(),
785  this_group_cutting_primitives.end());
786  }
787  }
788 }
