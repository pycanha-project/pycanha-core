198 void add_configure_geometry_item(

200 const std::string name = item->get_name();
201
202 if (_names.find(name) != _names.end()) {

230 void add_configure_geometry_group(
231 const std::shared_ptr<GeometryGroup>& item) {
232 const std::string name = item->get_name();
233
234 if (_names.find(name) != _names.end()) {
235 throw std::runtime_error("Name already exists.");
236 }
237
238 // To create a GeometryGroup within the model, the geometries
239 // need to be already in the model in the root node
240 // otherwise an exception is thrown
241 // This new group is added to the root node, and the geometries
242 // are moved from the root node to the this group
243
244 // Check if all the geometries are in the root node
245 // TODO: Implement geometry iterator like:
246 // for (auto geo_ptr : item->get_geometries()) {...}
247 for (const auto& geoitem : item->get_geometry_items()) {
248 const bool found = std::any_of(
249 _root_geometry_group->get_geometry_items().begin(),
250 _root_geometry_group->get_geometry_items().end(),
251 [&geoitem](const auto& root_geoitem) {
252 return geoitem->get_id() == root_geoitem->get_id();
253 });
254
255 if (!found) {
256 throw std::runtime_error(
257 "Geometry not in the root node. Cannot create group.");
258 }
259 }
260
261 for (const auto& geogroup : item->get_geometry_groups()) {
262 const bool found = std::any_of(
263 _root_geometry_group->get_geometry_groups().begin(),
264 _root_geometry_group->get_geometry_groups().end(),
265 [&geogroup](const auto& root_geogroup) {
266 return geogroup->get_id() == root_geogroup->get_id();
267 });
268
269 if (!found) {
270 throw std::runtime_error(
271 "Geometry not in the root node. Cannot create group.");
272 }
273 }
274
275 // Add the group to the root node
276
277 // First check if item is GeometryGroup or GeometryGroupCutted
278 const std::shared_ptr<GeometryGroupCutted> item_geometry_group_cutted =
279 std::dynamic_pointer_cast<GeometryGroupCutted>(item);
280 if (item_geometry_group_cutted) {
281 // `item` is a GeometryGroupCutted
282 _root_geometry_group->add_geometry_group_cutted(
283 item_geometry_group_cutted);
284 } else {
285 // `item` is a GeometryGroup
286 _root_geometry_group->add_geometry_group(item);
287 }
288 item->set_parent(_root_geometry_group);
289 _names.insert(name);
290
291 // Add the item to the set of geometries
292 _geometries.insert(item);
293 // Add the item to the map of id - name
294 _geometry_id_to_name[item->get_id()] = item->get_name();
295
296 // Remove the geometries from the root node and change the parent
297 for (const auto& geoitem : item->get_geometry_items()) {
298 _root_geometry_group->remove_geometry_item(geoitem);
299 geoitem->set_parent(item);
300 }
301 for (const auto& geogroup : item->get_geometry_groups()) {
302 _root_geometry_group->remove_geometry_group(geogroup);
303 geogroup->set_parent(item);
304 }
305
306 // If item is a GeometryGroupCutted, iterate over all cutting
307 // primitives
308 const std::shared_ptr<GeometryGroupCutted> cutted_item =
309 std::dynamic_pointer_cast<GeometryGroupCutted>(item);
310 if (cutted_item) {
311 for (const auto& geocutteditem :
312 cutted_item->get_cutting_geometry_items()) {
313 _root_geometry_group->remove_geometry_item(geocutteditem);
314 geocutteditem->set_parent(item);
315 }
316 }
317 }
